# Simple Reflex Agent for AQI

def check_aqi(aqi):
    
    if aqi <= 50:
        return "Good"
    elif aqi <= 100:
        return "Moderate"
    elif aqi <= 150:
        return "Unhealthy for Sensitive Groups"
    elif aqi <= 200:
        return "Unhealthy"
    elif aqi <= 300:
        return "Very Unhealthy"
    else:
        return "Hazardous"


# read sensor data file
file = open("sensor.txt","r")

for line in file:
    aqi = int(line.strip())
    print("AQI:",aqi,"Category:",check_aqi(aqi))

file.close()
